#Check if a number is a prime number.

num = int(input("Enter a number: "))
if num > 1:
    for i in range(2, num):
        if num % i == 0:
            print("Not prime")
            break
    else:
        print("Prime")
else:
    print("Not prime")

#Write a python program to check whether a number is in a given range.

number = float(input("Enter a number: "))
start = float(input("Enter the start of the range: "))
end = float(input("Enter the end of the range: "))

if start <= number <= end:
    print("The number is in the range.")
else:
    print("The number is not in the range.")

#Calculate the number of upper case letters and lower case letters in string.

s = input("Enter a string: ")
print("Upper case:", sum(1 for c in s if c.isupper()))
print("Lower case:", sum(1 for c in s if c.islower()))
